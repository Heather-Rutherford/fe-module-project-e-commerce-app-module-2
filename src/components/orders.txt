// Function to add order to Firestore
async function addOrder(orderData) {
  try {
    const docRef = await addDoc(collection(db, 'orders'), orderData);
    console.log('Order placed with ID:', docRef.id);
  } catch (error) {
    console.error('Error adding order:', error);
  }
}

how to query for orders based on the suggested example

To query for orders based on the suggested Firestore structure, you can use Firestore’s query functions. For example, to get all orders for a specific user:

import { db } from '../firebaseConfig';
import { collection, query, where, getDocs } from 'firebase/firestore';

async function getOrdersByUser(userId) {
  const ordersRef = collection(db, 'orders');
  const q = query(ordersRef, where('userId', '==', userId));
  const querySnapshot = await getDocs(q);
  const orders = [];
  querySnapshot.forEach((doc) => {
    orders.push({ id: doc.id, ...doc.data() });
  });
  return orders;
}
